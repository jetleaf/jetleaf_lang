bool notEqualsAll(List<String> values, {bool isLowerCase = false, bool isUpperCase = false, bool isIgnoreCase = false}) {
  if(isUpperCase) {
    return !values.all((v) => v.toUpperCase().equals(toUpperCase()));
  }

  return !values.all((v) => equalsIgnoreCase(v));
}
