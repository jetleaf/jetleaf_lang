@override
Map<K2, V2> map<K2, V2>(MapEntry<K2, V2> Function(K key, V value) convert) {
  final newMap = HashMap<K2, V2>();
  forEach((key, value) {
    final entry = convert(key, value);
    newMap[entry.key] = entry.value;
  });
  return newMap;
}
